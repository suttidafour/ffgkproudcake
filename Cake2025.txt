  const data = snapshot.val();
  createSignElement(data.text, data.color);
});

// ===== Create sign element =====
function createSignElement(text, color) {
  const sign = document.createElement("div");
  sign.className = "sign animate";
  const label = document.createElement("div");
  label.className = "label";
  label.innerText = text;
  label.style.background = color;
  const stick = document.createElement("div");
  stick.className = "stick";
  sign.append(label, stick);
  sign.style.left = "50%";
  sign.style.top = "40px";
  sign.style.transform = "translateX(-50%)";
  enableDrag(sign);
  signsBox.appendChild(sign);
  cake.classList.add("shake");
  setTimeout(() => cake.classList.remove("shake"), 350);
  setTimeout(() => sign.classList.remove("animate"), 600);
}

// ===== Drag =====
function enableDrag(el) {
  let offsetX, offsetY, dragging = false;
  el.addEventListener("pointerdown", e => { dragging = true; el.setPointerCapture(e.pointerId); const r = el.getBoundingClientRect(); offsetX = e.clientX - r.left; offsetY = e.clientY - r.top; });
  el.addEventListener("pointermove", e => {
    if (!dragging) return;
    const box = signsBox.getBoundingClientRect();
    let x = e.clientX - box.left - offsetX;
    let y = e.clientY - box.top - offsetY;
    x = Math.max(0, Math.min(x, box.width - el.offsetWidth));
    y = Math.max(0, Math.min(y, box.height - el.offsetHeight));
    el.style.left = x + "px";
    el.style.top = y + "px";
    el.style.transform = "";
  });
  el.addEventListener("pointerup", () => dragging = false);
}
